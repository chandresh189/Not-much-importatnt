
/*   prime number                                                                            */


#include<iostream>
using namespace std;
int main()
{
int n,i,c=0;
cin>>n;
for(i=15;i<=n;i++)
	{
	  if(n % i == 0)
		 {
		 c++;
		 }
	}
   if(c == 2)
	{
	 cout<<"yes"<<endl;
	}
    else{
      cout<<"no"<<endl;	
	}
		
	return 0;
}
