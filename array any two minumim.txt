#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
	int i,j,n;
		cin>>n;
		int arr[n];
		for(j=0;j<n;j++)
		{
			cin>>arr[j];
		}
		sort(arr, arr+n);
		int minn = INT_MAX;
		for(j=1;j<n;j++)
		{
		  minn = min(arr[j]-arr[j-1],minn);	
		}
		cout << minn << "\n";
	}

return 0;
}
