class Solution {
public:
    bool detectCapitalUse(string word) {
        int caps = 0;
        for(auto it : word){
            if(it >='A' && it <= 'Z')  caps++;
        }
        if(caps == word.size() || caps == 0)  return true;
        else if((caps == 1) && (isupper(word[0]))) return true;
        else return false;
    }
};
