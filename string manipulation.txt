******Remove duplicates from a string*******
 

int main() 
{ 
	string str = "aaaaabbbbbb"; 
	sort(str.begin(), str.end()); 

	// Using unique() method 
	auto res = unique(str.begin(), str.end()); 

	cout << string(str.begin(), res) << endl; 
} 
------------------------
Input: Geeks for geeks
Output: Gefgkors
